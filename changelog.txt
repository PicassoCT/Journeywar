float PFormula::evaluateFormula(PObject * object, unsigned int timeInMsSinceLastCall, float Numeral, int Mass)
{
float Result=0;

	for (int i=0;i  < Formula.size;i++)
	{
		switch(Formula[i].FollowUpOperation):
		{
			case (PLUS):
			{
				Result+= pow(Formula[i].FollowUpOperation.Scalar* getProvidedVar(Formula[i].FollowUpOperation.Variable, object),Formula[i].Exponent);
			}
			case (MINUS):
			{
				Result-= pow(Formula[i].FollowUpOperation.Scalar* getProvidedVar(Formula[i].FollowUpOperation.Variable, object),Formula[i].Exponent);
			}
			case (MULTIPLICTION):
			{
				Result*= pow(Formula[i].FollowUpOperation.Scalar* getProvidedVar(Formula[i].FollowUpOperation.Variable, object),Formula[i].Exponent);
			}
			case (DIVISION):
			{
				Result/= pow(Formula[i].FollowUpOperation.Scalar* getProvidedVar(Formula[i].FollowUpOperation.Variable, object),Formula[i].Exponent);
			}
			case (NOP):
			{
				return Result;
			}
			case default:
			{
				return Result;
			}
		}
	}


}
